# install typescript:

tsc --init
# edit file tsconfig.json, di dalam compilerOptions, tambahkan 2 baris kode ini:

"outDir": "./scripts",

# buat sebuah file misalnya demo.ts di dalam folder 'src'
# pres Ctrl+Shift+B and choose tsc:watch-tsconfig.json
